stockVol= Dialog.getNumber();

stockVolPref = "";
finalVolPref = "";

stockConcScale=1;
stockVolScale=1;
finalConcScale = 1;
finalVolScale=1;

if (stockVolUnits==" ml")
	stockVolScale= 1000;
if (stockVolUnits==" 痞")
	stockVolScale= 1000000;

	stockConcScale= 1000;

	stockVol=((finalVol/finalVolScale)*(finalConc/finalConcScale))/(stockConc/stockConcScale);
